blitzdg
an open-source project aiming to implement parallel discontinuous Galerkin (dg) solvers for common partial differential equations systems using blitz++ for array and tensor manipulations and MPI for distributed parallelism.

24  bool operator()(const real_vector_type & in, real_vector_type & out) {

26 
27  blitz::firstIndex ii;
28  blitz::secondIndex jj;
29  blitz::thirdIndex kk;
30 
31  const index_type numFaces = dg.numFaces();
32  const index_type K = dg.numElements();
33  const index_type Nfp = dg.numFacePoints();
34  const index_type Np = dg.numLocalPoints();
35 
36  real_vector_type du(numFaces*Nfp*K);
37  real_vector_type uM(numFaces*Nfp*K);
38  real_vector_type uP(numFaces*Nfp*K);
39  real_vector_type nxVec(numFaces*Nfp*K);
40  real_vector_type nyVec(numFaces*Nfp*K);

42 
43  const bool byRowsOpt = false;
44  blitzdg::vectorToFull(in, u, byRowsOpt);
45  blitzdg::fullToVector(dg.nx(), nxVec, byRowsOpt);
46  blitzdg::fullToVector(dg.ny(), nyVec, byRowsOpt);
47 
48  real_matrix_type ux(Np,K), uy(Np,K);
49  real_vector_type uVec(Np*K), uxVec(Np*K), uyVec(Np*K);
50 
51  blitzdg::fullToVector(u, uVec, byRowsOpt);
52 
53  const real_matrix_type& Dr = dg.Dr(), Ds = dg.Ds();
54  real_matrix_type dudr(Np,K), duds(Np,K);
55 
56  dudr = blitz::sum(Dr(ii,kk)*u(kk,jj), kk);
57  duds = blitz::sum(Ds(ii,kk)*u(kk,jj), kk);
58 
59  ux = dg.rx()*dudr + dg.sx()*duds;
60  uy = dg.ry()*dudr + dg.sy()*duds;
61 
62  blitzdg::fullToVector(ux, uxVec, byRowsOpt);
63  blitzdg::fullToVector(uy, uyVec, byRowsOpt);
64 
65  blitzdg::applyIndexMap(uVec, dg.vmapM(), uM);
66  blitzdg::applyIndexMap(uVec, dg.vmapP(), uP);
67 
68  // impose boundary condition -- Dirichlet BC's (reflective)
69  //const index_hashmap& bcMap = dg.bcmap();
70  std::vector<index_type> mapD = dg.bcmap().at(BCTag::Wall); // forcing dirichlet for now.
71  for (index_type i=0; i < static_cast<int>(mapD.size()); ++i) {
72  index_type d = mapD[i];
73  uP(d) = -uM(d);
74  }
75 
76  // impose boundary condition -- Neuman BC's
77  // nothing to do => uP(mapN) = uM(mapN) by default.
78 
79  du = uM - uP;
80  real_matrix_type duMat(Nfp*numFaces, K);
81  blitzdg::vectorToFull(du, duMat, byRowsOpt);
82 
83  // Auxiliary vector variable to hold first derivative (gradient).
84  const real_matrix_type& Lift = dg.lift();
85  real_matrix_type qx(Np, K), qy(Np, K);
86  qx = ux;
87  qy = uy;
88 
89  real_matrix_type surfaceRHSx(Nfp*numFaces, K), surfaceRHSy(Nfp*numFaces, K);
90  surfaceRHSx = dg.fscale()*(dg.nx()*duMat/2.0);
91  surfaceRHSy = dg.fscale()*(dg.ny()*duMat/2.0);
92 
93  qx -= blitz::sum(Lift(ii,kk)*surfaceRHSx(kk,jj), kk);
94  qy -= blitz::sum(Lift(ii,kk)*surfaceRHSy(kk,jj), kk);
95 
96  real_vector_type qxVec(Np*K), qyVec(Np*K);
97 
98  blitzdg::fullToVector(qx, qxVec, byRowsOpt);
99  blitzdg::fullToVector(qy, qyVec, byRowsOpt);
100 
101  real_vector_type dqx(numFaces*Nfp*K), qxM(numFaces*Nfp*K), qxP(numFaces*Nfp*K);
102  real_vector_type dqy(numFaces*Nfp*K), qyM(numFaces*Nfp*K), qyP(numFaces*Nfp*K);
103  real_vector_type uxM(numFaces*Nfp*K), uxP(numFaces*Nfp*K);
104  real_vector_type uyM(numFaces*Nfp*K), uyP(numFaces*Nfp*K);
105 
106  blitzdg::applyIndexMap(qxVec, dg.vmapM(), qxM);
107  blitzdg::applyIndexMap(qxVec, dg.vmapP(), qxP);
108 
109  blitzdg::applyIndexMap(qyVec, dg.vmapM(), qyM);
110  blitzdg::applyIndexMap(qyVec, dg.vmapP(), qyP);
111 
112  blitzdg::applyIndexMap(uxVec, dg.vmapM(), uxM);
113  blitzdg::applyIndexMap(uxVec, dg.vmapP(), uxP);
114 
115  blitzdg::applyIndexMap(uyVec, dg.vmapM(), uyM);
116  blitzdg::applyIndexMap(uyVec, dg.vmapP(), uyP);
117 
118  // Neuman BC's imply auxiliary vector "(qx,qy) = 0" on the boundary
119  //std::vector<index_type> mapN = dg.bcmap().at(BCTag::Neuman);
120  std::vector<index_type> mapN;
121  if (dg.bcmap().count( BCTag::Neuman )) {
122  mapN = dg.bcmap().at(BCTag::Neuman);
123  }
124 
125  for (index_type i=0; i < static_cast<int>(mapN.size()); ++i) {
126  index_type n = mapN[i];
127  uxP(n) = uxM(n) - 2*nxVec(n)*(uxM(n)*nxVec(n) + uyM(n)*nyVec(n));
128  uyP(n) = uyM(n) - 2*nyVec(n)*(uxM(n)*nxVec(n) + uyM(n)*nyVec(n));
129  }
130 
131  // the weirdness that is interior penalty....
132  qxP = 0.5*(uxM + uxP);
133  qyP = 0.5*(uyM + uyP);
134 
135  dqx = qxM - qxP;
136  dqy = qyM - qyP;
137 
138  real_matrix_type dqxMat(Nfp*numFaces, K), dqyMat(Nfp*numFaces, K);
139  blitzdg::vectorToFull(dqx, dqxMat, byRowsOpt);
140  blitzdg::vectorToFull(dqy, dqyMat, byRowsOpt);
141 
142  // compute penalty scaling.
143  const double hmin = 2.0/ blitzdg::normMax(dg.jacobian());
144  const double tau = Np/hmin;
145 
146  real_matrix_type surfaceRHS(Nfp*numFaces, K);
147 
148  surfaceRHS = (dg.nx()*dqxMat + dg.ny()*dqyMat + tau*duMat)/2.0; // x component
149 
150  real_matrix_type MassMatrix(Np,Np), tmp(Np,K), result(Np,K);
151 
152  const real_matrix_type& Vinv = dg.Vinv();
153  MassMatrix = blitz::sum(Vinv(kk,ii)*Vinv(kk,jj), kk); // local one.
154 
155  // Commpute Laplacian.
156  tmp = dg.rx()*blitz::sum(Dr(ii,kk)*qx(kk,jj), kk) + dg.sx()*blitz::sum(Ds(ii,kk)*qx(kk,jj), kk);
157  tmp += dg.ry()*blitz::sum(Dr(ii,kk)*qy(kk,jj), kk) + dg.sy()*blitz::sum(Ds(ii,kk)*qy(kk,jj), kk);
158  tmp -= sum(Lift(ii,kk)*surfaceRHS(kk,jj), kk);
159 
160  // Multiply by mass matrix for a symmetric problem.
161  //result = dg.jacobian()*(sum(MassMatrix(ii,kk)*tmp(kk,jj),kk));
162  result = tmp;
163 
164  blitzdg::fullToVector(result, out, byRowsOpt);
165 
166  return true;
167  }
