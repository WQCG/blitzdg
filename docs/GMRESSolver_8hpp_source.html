150  GMRESOut GMRESSolver::solve(Matvec mvec, Precon precon, const vector_type& b,
151  vector_type& x, const GMRESParams& params) const
152  {
153  checkGMRESParams(params); // check for any errors in input parameters
154  const index_type N = x.size();
155  const index_type kspaceSz = std::min(N, params.kspaceSz);
156  vector_type r(N), w(N), pv(N), s(kspaceSz + 1);
157  matrix_type H(kspaceSz + 1, kspaceSz, blitz::ColumnMajorArray<2>());
158  matrix_type V(N, kspaceSz + 1, blitz::ColumnMajorArray<2>());
159  std::vector<std::pair<real_type, real_type>> givens(kspaceSz);
160  GMRESOut ret;
161  r = b;
162 
163  real_type bnrm = norm2(b);
164  real_type rnrm = bnrm;
165  if (bnrm == real_type(0)) {
166  bnrm = real_type(1);
167  }
168  const real_type bnrmInv = real_type(1) / bnrm;
169 
170  // if initial guess is nonzero, update residual
171  if (details::isNonzero(x)) {
172  if (!mvec(x, w)) {
173  ret.flag = ConvFlag::matvec_fail;
174  ret.msg = "matvec failed when computing initial residual";
175  return ret;
176  }
177  r -= w;
178  rnrm = norm2(r);
179  }
180 
181  // check for inf or nan
182  if (!std::isfinite(rnrm)) {
183  ret.flag = ConvFlag::inf_or_nan;
184  return ret;
185  }
186 
187  // define convergence and divergence tolerances
188  const real_type convTol = std::max(params.relTol * bnrm, params.absTol);
189  const real_type divTol = params.divTol * rnrm;
190 
191  // check for early termination
192  if (rnrm <= convTol) {
193  ret.flag = ConvFlag::success;
194  ret.relres = bnrmInv * rnrm;
195  return ret;
196  }
197 
198  if (params.verbose) {
199  std::cout << "inner iter outer iter relative rnrm conv rate\n";
200  std::cout << std::setw(10) << 0 << " "
201  << std::setw(10) << 0 << " "
202  << std::scientific << std::setprecision(2)
203  << std::setw(13) << bnrmInv * rnrm << " "
204  << std::scientific << std::setprecision(2)
205  << std::setw(9) << 1 << std::endl;
206  }
207 
208  // time to grip it and rip it
209  bool testTrueRnrm = params.testTrueRnrm;
210  for (index_type k = 0; k < params.maxits; ++k) {
211  ret.relres = bnrmInv * rnrm;
212  ret.outerIts = k;
213  real_type scal = real_type(1) / rnrm;
214  V(blitz::Range::all(), 0) = r;
215  V(blitz::Range::all(), 0) *= scal;
216  s = real_type(0);
217  s(0) = rnrm;
218  index_type iupdate = kspaceSz;
219  for (index_type i = 0; i < kspaceSz; ++i) {
220  ret.innerIts = i;
221  real_type rnrmold = rnrm;
222  vector_type viref = V(blitz::Range::all(), i); // ref to slice
223  if (!precon(viref, pv)) { // pv = M^{-1}*V(:,i)
224  ret.flag = ConvFlag::precon_fail;
225  ret.msg = "preconditioner application in inner loop failed";
226  return ret;
227  }
228  if (!mvec(pv, w)) { // w = A*pv = A*M^{-1}*V(:,i)
229  ret.flag = ConvFlag::matvec_fail;
230  ret.msg = "matvec in inner loop failed";
231  return ret;
232  }
233  // MGS loop
234  for (index_type j = 0; j <= i; ++j) {
235  real_type hji = sum(w * V(blitz::Range::all(), j));
236  w -= hji * V(blitz::Range::all(), j);
237  H(j, i) = hji;
238  }
239  H(i + 1, i) = norm2(w);
240 
241  // If H(i + 1, i) = 0, then the ith Givens rotation J_i
242  // is equal to the identity matrix and hence the
243  // (i + 1)th element of J_i*s is equal to zero, which
244  // will trigger termination of the inner loop.
245  if (H(i + 1, i) != real_type(0)) {
246  V(blitz::Range::all(), i + 1) = w;
247  scal = real_type(1) / H(i + 1, i);
248  V(blitz::Range::all(), i + 1) *= scal;
249  }
250 
251  // apply Givens rotations 0,...,i-1 to column i of H
252  for (index_type j = 0; j < i; ++j) {
253  applyGivens(givens[j].first, givens[j].second, H(j, i), H(j + 1, i));
254  }
255 
256  // compute the ith Givens rotation and apply it to column i of H
257  DROTG(H(i, i), H(i + 1, i), givens[i].first, givens[i].second);
258 
259  // check if H is singular, i.e., H(i,i) = 0
260  if (H(i, i) == real_type(0)) {
261  ret.flag = ConvFlag::breakdown;
262  std::ostringstream ss;
263  ss << "H(" << i << "," << i << ") = 0";
264  ret.msg = ss.str();
265  return ret;
266  }
267 
268  // apply the ith Givens rotation to s and get the approximate residual norm
269  applyGivens(givens[i].first, givens[i].second, s(i), s(i + 1));
270  rnrm = std::abs(s(i + 1));
271 
272  // check for inf or nan
273  if (!std::isfinite(rnrm)) {
274  ret.flag = ConvFlag::inf_or_nan;
275  return ret;
276  }
277 
278  if (params.verbose) {
279  std::cout << std::setw(10) << k + 1 << " "
280  << std::setw(10) << i + 1 << " "
281  << std::scientific << std::setprecision(2)
282  << std::setw(13) << bnrmInv * rnrm << " "
283  << std::scientific << std::setprecision(2)
284  << std::setw(9) << rnrm / rnrmold << std::endl;
285  rnrmold = rnrm;
286  }
287 
288  // check for convergence
289  if (rnrm <= convTol) {
290  ret.flag = ConvFlag::success;
291  iupdate = i + 1;
292  break;
293  }
294 
295  // check for divergence
296  if (rnrm >= divTol) {
297  ret.flag = ConvFlag::diverged;
298  iupdate = i + 1;
299  break;
300  }
301  }
302  // update the solution
303  // solve H(0:iupdate-1,0:iupdate-1)*y = s(0:iupdate-1)
304  details::backSolve(iupdate, H, s);
305  // compute w = V(:,0:iupdate-1)*y
306  details::matTimesVec(iupdate, V, s, w);
307  // compute pv = M^{-1}*w
308  if (!precon(w, pv)) {
309  ret.flag = ConvFlag::precon_fail;
310  ret.msg = "preconditioner application failed when updating solution";
311  return ret;
312  }
313  // update x
314  x += pv;
315 
316  // if termination signaled in inner loop, then exit
317  if (ret.flag != ConvFlag::unconverged)
318  break;
319 
320  // update the residual
321  if (!mvec(x, w)) {
322  ret.flag = ConvFlag::matvec_fail;
323  ret.msg = "matvec failed when updating residual";
324  return ret;
325  }
326  r = b - w;
327  rnrm = norm2(r);
328 
329  // check for inf or nan
330  if (!std::isfinite(rnrm)) {
331  ret.flag = ConvFlag::inf_or_nan;
332  return ret;
333  }
334 
335  // check for convergence
336  if (rnrm <= convTol) {
337  ret.flag = ConvFlag::success;
338  testTrueRnrm = false;
339  break;
340  }
341 
342  // check for divergence
343  if (rnrm >= divTol) {
344  ret.flag = ConvFlag::diverged;
345  break;
346  }
347 
348  // check for stagnation
349  bool stagnated = true;
350  for (index_type j = 0; j < N; ++j) {
351  if (x(j) != real_type(0) &&
352  std::abs(pv(j)) > params.stgTol * std::abs(x(j)))
353  {
354  stagnated = false;
355  break;
356  }
357  }
358  if (stagnated) {
359  ret.flag = ConvFlag::stagnation;
360  break;
361  }
362  }
363  if (ret.flag == ConvFlag::unconverged)
364  ret.flag = ConvFlag::maxits;
365 
366  // if requested, check that the true residual norm satisfies
367  // the convergence criteria
368  if (ret.flag == ConvFlag::success && testTrueRnrm) {
369  if (mvec(x, w)) {
370  r = b - w;
371  rnrm = norm2(r);
372  if (std::isfinite(rnrm) && rnrm > convTol)
373  ret.flag = ConvFlag::true_rnrm;
374  }
375  else
376  ret.msg = "matvec failed when computing true residual -> test skipped";
377  }
378  ret.relres = bnrmInv * rnrm;
379  return ret;
380  }
