217  vector_type r(N), w(N), pv(N), s(kspaceSz + 1);
218  matrix_type H(kspaceSz + 1, kspaceSz, blitz::ColumnMajorArray<2>());
219  matrix_type V(N, kspaceSz + 1, blitz::ColumnMajorArray<2>());
220  std::vector<std::pair<real_type, real_type>> givens(kspaceSz);
221  GMRESOut ret;
222  r = b;
223 
224  real_type bnrm = norm2(b);
225  real_type rnrm = bnrm;
226  if (bnrm == real_type(0)) {
227  bnrm = real_type(1);
228  }
229  const real_type bnrmInv = real_type(1) / bnrm;
230 
231  // if initial guess is nonzero, update residual
232  if (details::isNonzero(x)) {
233  if (!mvec(x, w)) {
234  ret.flag = ConvFlag::matvec_fail;
235  ret.msg = "matvec failed when computing initial residual";
236  return ret;
237  }
238  r -= w;
239  rnrm = norm2(r);
240  }
241

243  if (!std::isfinite(rnrm)) {
244  ret.flag = ConvFlag::inf_or_nan;
245  return ret;
246  }
247

249  const real_type convTol = std::max(params.relTol * bnrm, params.absTol);
250  const real_type divTol = params.divTol * rnrm;
251 
252  // check for early termination
253  if (rnrm <= convTol) {
254  ret.flag = ConvFlag::success;
255  ret.relres = bnrmInv * rnrm;
256  return ret;
257  }
258 
259  if (params.verbose) {
260  std::cout << "inner iter outer iter relative rnrm conv rate\n";
261  std::cout << std::setw(10) << 0 << " "
262  << std::setw(10) << 0 << " "
263  << std::scientific << std::setprecision(2)
264  << std::setw(13) << bnrmInv * rnrm << " "
265  << std::scientific << std::setprecision(2)
266  << std::setw(9) << 1 << std::endl;
267  }

271  for (index_type k = 0; k < params.maxits; ++k) {
272  ret.relres = bnrmInv * rnrm;
273  ret.outerIts = k;
274  real_type scal = real_type(1) / rnrm;
275  V(blitz::Range::all(), 0) = r;
276  V(blitz::Range::all(), 0) *= scal;
277  s = real_type(0);
278  s(0) = rnrm;
279  index_type iupdate = kspaceSz;
280  for (index_type i = 0; i < kspaceSz; ++i) {
281  ret.innerIts = i;
282  real_type rnrmold = rnrm;
283  vector_type viref = V(blitz::Range::all(), i); // ref to slice
284  if (!precon(viref, pv)) { // pv = M^{-1}*V(:,i)
285  ret.flag = ConvFlag::precon_fail;
286  ret.msg = "preconditioner application in inner loop failed";
287  return ret;
288  }
289  if (!mvec(pv, w)) { // w = A*pv = A*M^{-1}*V(:,i)
290  ret.flag = ConvFlag::matvec_fail;
291  ret.msg = "matvec in inner loop failed";
292  return ret;
293  }
294  // MGS loop
295  for (index_type j = 0; j <= i; ++j) {
296  real_type hji = sum(w * V(blitz::Range::all(), j));
297  w -= hji * V(blitz::Range::all(), j);
298  H(j, i) = hji;
299  }
300  H(i + 1, i) = norm2(w);
301 
302  // If H(i + 1, i) = 0, then the ith Givens rotation J_i
303  // is equal to the identity matrix and hence the
304  // (i + 1)th element of J_i*s is equal to zero, which
305  // will trigger termination of the inner loop.
306  if (H(i + 1, i) != real_type(0)) {
307  V(blitz::Range::all(), i + 1) = w;
308  scal = real_type(1) / H(i + 1, i);
309  V(blitz::Range::all(), i + 1) *= scal;
310  }
311 
312  // apply Givens rotations 0,...,i-1 to column i of H
313  for (index_type j = 0; j < i; ++j) {
314  applyGivens(givens[j].first, givens[j].second, H(j, i), H(j + 1, i));
315  }
316 
317  // compute the ith Givens rotation and apply it to column i of H
318  DROTG(H(i, i), H(i + 1, i), givens[i].first, givens[i].second);
319 
320  // check if H is singular, i.e., H(i,i) = 0
321  if (H(i, i) == real_type(0)) {
322  ret.flag = ConvFlag::breakdown;
323  std::ostringstream ss;
324  ss << "H(" << i << "," << i << ") = 0";
325  ret.msg = ss.str();
326  return ret;
327  }
328 
329  // apply the ith Givens rotation to s and get the approximate residual norm
330  applyGivens(givens[i].first, givens[i].second, s(i), s(i + 1));
331  rnrm = std::abs(s(i + 1));
332 
333  // check for inf or nan
334  if (!std::isfinite(rnrm)) {
335  ret.flag = ConvFlag::inf_or_nan;
336  return ret;
337  }
338

341  << std::setw(10) << i + 1 << " "
342  << std::scientific << std::setprecision(2)
343  << std::setw(13) << bnrmInv * rnrm << " "
344  << std::scientific << std::setprecision(2)

346  rnrmold = rnrm;
347  }

352  iupdate = i + 1;
353  break;
354  }
355

357  if (rnrm >= divTol) {
358  ret.flag = ConvFlag::diverged;
359  iupdate = i + 1;
360  break;
361  }
362  }
363  // update the solution
364  // solve H(0:iupdate-1,0:iupdate-1)*y = s(0:iupdate-1)
365  details::backSolve(iupdate, H, s);
366  // compute w = V(:,0:iupdate-1)*y
367  details::matTimesVec(iupdate, V, s, w);
368  // compute pv = M^{-1}*w
369  if (!precon(w, pv)) {
370  ret.flag = ConvFlag::precon_fail;
371  ret.msg = "preconditioner application failed when updating solution";
372  return ret;
373  }
374  // update x
375  x += pv;
376 
377  // if termination signaled in inner loop, then exit
378  if (ret.flag != ConvFlag::unconverged)
379  break;
380 
381  // update the residual
382  if (!mvec(x, w)) {
383  ret.flag = ConvFlag::matvec_fail;
384  ret.msg = "matvec failed when updating residual";
385  return ret;
386  }
387  r = b - w;
388  rnrm = norm2(r);
389 
390  // check for inf or nan
391  if (!std::isfinite(rnrm)) {
392  ret.flag = ConvFlag::inf_or_nan;
393  return ret;
394  }
395 
396  // check for convergence

398  ret.flag = ConvFlag::success;
399  testTrueRnrm = false;
400  break;
401  }
402 
403  // check for divergence
404  if (rnrm >= divTol) {
405  ret.flag = ConvFlag::diverged;
406  break;
407  }

412  if (x(j) != real_type(0) &&
413  std::abs(pv(j)) > params.stgTol * std::abs(x(j)))
414  {

416  break;
417  }
418  }

421  break;
422  }
423  }

426

428  // the convergence criteria
429  if (ret.flag == ConvFlag::success && testTrueRnrm) {
430  if (mvec(x, w)) {
431  r = b - w;

433  if (std::isfinite(rnrm) && rnrm > convTol)
434  ret.flag = ConvFlag::true_rnrm;
435  }

438  }
