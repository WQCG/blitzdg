blitzdg
an open-source project aiming to implement parallel discontinuous Galerkin (dg) solvers for common partial differential equations systems using blitz++ for array and tensor manipulations and MPI for distributed parallelism.

17  class PoissonOperator {
18  const Nodes1DProvisioner* nodes1D;
19  public:
20  explicit PoissonOperator(const Nodes1DProvisioner& _nodes1D)
21  : nodes1D{ &_nodes1D }
22  {}
23 
24  bool operator()(const real_vector_type & in, real_vector_type & out) {
25  // Blitz indices
26  blitz::firstIndex ii;
27  blitz::secondIndex jj;
28  blitz::thirdIndex kk;
29  const real_matrix_type& Dr = nodes1D->get_Dr();
30  const real_matrix_type& rx = nodes1D->get_rx();
31  const real_matrix_type& J = nodes1D->get_J();
32  const real_matrix_type& Lift = nodes1D->get_Lift();
33  const real_matrix_type& Fscale = nodes1D->get_Fscale();
34  const real_matrix_type& nx = nodes1D->get_nx();
35  const real_matrix_type& Vinv = nodes1D->get_Vinv();
36 
37  // Get volume to surface maps.
38  const index_vector_type& vmapM = nodes1D->get_vmapM();
39  const index_vector_type& vmapP = nodes1D->get_vmapP();
40 
41  // boundary indices.
42  index_type mapO = nodes1D->get_mapO();
43  index_type mapI = nodes1D->get_mapI();
44 
45  index_type numFaces = nodes1D->NumFaces;
46  index_type Nfp = nodes1D->NumFacePoints;
47  index_type Np = nodes1D->get_NumLocalPoints();
48  index_type K = nodes1D->get_NumElements();
49 
50  real_matrix_type VinvTrans(Np, Np);
51  VinvTrans = Vinv(jj,ii);
52 
53  real_vector_type du(numFaces*Nfp*K);
54  real_vector_type uM(numFaces*Nfp*K);
55  real_vector_type uP(numFaces*Nfp*K);
56  real_vector_type nxVec(numFaces*Nfp*K);
57  real_matrix_type u(Np, K);
58 
59  const bool byRowsOpt = false;
60  blitzdg::vectorToFull(in, u, byRowsOpt);
61  blitzdg::fullToVector(nx, nxVec, byRowsOpt);
62 
63  real_matrix_type ux(Np, K);
64  real_vector_type uVec(Np*K);
65  real_vector_type uxVec(Np*K);
66 
67  blitzdg::fullToVector(u, uVec, byRowsOpt);
68 
69  ux = rx*sum(Dr(ii,kk)*u(kk,jj), kk);
70  blitzdg::fullToVector(ux, uxVec, byRowsOpt);
71 
72  blitzdg::applyIndexMap(uVec, vmapM, uM);
73  blitzdg::applyIndexMap(uVec, vmapP, uP);
74 
75  // impose boundary condition -- Dirichlet BC's
76  uP(mapI) = -uM(mapI);
77  uP(mapO) = -uM(mapO);
78 
79  // Compute jump in flux:
80  du = (uM - uP);
81 
82  real_matrix_type duMat(Nfp*numFaces, K);
83  blitzdg::vectorToFull(du, duMat, byRowsOpt);
84 
85  // Auxiliary variable to hold first derivative.
86  real_matrix_type q(Np, K);
87  q = rx*sum(Dr(ii,kk)*u(kk,jj), kk);
88 
89  real_matrix_type surfaceRHS(Nfp*numFaces, K);
90  real_vector_type qVec(Np*K);
91 
92  surfaceRHS = Fscale*(nx*duMat/2.0);
93 
94  q = rx*(sum(Dr(ii,kk)*u(kk,jj), kk));
95  q -= sum(Lift(ii,kk)*(surfaceRHS(kk,jj)), kk);
96 
97  blitzdg::fullToVector(q, qVec, byRowsOpt);
98  real_vector_type dq(numFaces*Nfp*K);
99  real_vector_type qM(numFaces*Nfp*K);
100  real_vector_type qP(numFaces*Nfp*K);
101  real_vector_type uxM(numFaces*Nfp*K);
102  real_vector_type uxP(numFaces*Nfp*K);
103 
104  blitzdg::applyIndexMap(qVec, vmapM, qM);
105  blitzdg::applyIndexMap(uxVec, vmapM, uxM);
106  blitzdg::applyIndexMap(uxVec, vmapP, uxP);
107 
108  // Neumann BC's on q
109  uxP(mapI) = uxM(mapI);
110  uxP(mapO) = uxM(mapO);
111 
112  qP = 0.5*(uxM + uxP);
113 
114  dq = qM -qP;
115 
116  real_matrix_type dqMat(Nfp*numFaces, K);
117  blitzdg::vectorToFull(dq, dqMat, byRowsOpt);
118 
119  const double hmin = 2.0/blitzdg::normMax(rx);
120  const double tau = (Np*Np)/hmin;
121 
122  surfaceRHS = nx*(dqMat + tau*nx*duMat);
123 
124  real_matrix_type MassMatrix(Np, Np);
125  MassMatrix = sum(Vinv(kk,ii)*Vinv(kk,jj), kk);
126 
127  real_matrix_type tmp(Np, K);
128  real_matrix_type result(Np, K);
129 
130  tmp = rx*(sum(Dr(ii,kk)*q(kk,jj), kk));
131  tmp -= sum(Lift(ii,kk)*(surfaceRHS(kk,jj)), kk);
132  result = J*(sum(MassMatrix(ii,kk)*tmp(kk,jj), kk));
133 
134  blitzdg::fullToVector(result, out, byRowsOpt);
135 
136  return true;
137  } // forward operator
138  };
